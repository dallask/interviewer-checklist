```
const handleExportYaml = () => {
  const s = useAppStore.getState();
  const currentSession = {
    version: 3,
    id: s.activeSessionId,
    scores: s.scores,
    overrides: s.overrides,
    notes: s.notes,
    topicNotes: s.topicNotes,
    customQuestions: s.customQuestions,
    candidate: s.candidate,
  } as const;
  const sessionName = s.manifest?.sessions.find(x => x.id === s.activeSessionId)?.name ?? '';
  const yaml = exportSession(currentSession as unknown as Parameters<typeof exportSession>[0], sessionName, DEFAULT_SECTIONS);
  downloadYaml(yaml, buildFilename(sessionName));
};

const handleOpenImportYaml = () => {
  setImportError(null);
  importFileInputRef.current?.click();
};

const handleImportFileChange = async (e: React.ChangeEvent<HTMLInputElement>) => {
  const file = e.target.files?.[0];
  if (!file) return;
  if (file.size > MAX_YAML_BYTES) {
    setImportError(`File too large (${file.size} bytes; max ${MAX_YAML_BYTES})`);
    return;
  }
  const text = await file.text();
  const parsed = parseYaml(text);
  if (!parsed.ok) {
    setImportError(`Invalid YAML: ${parsed.error}`);
    return;
  }
  const format = detectFormat(parsed.value);
  const preview = format === 'structural'
    ? parseStructural(parsed.value, DEFAULT_SECTIONS)
    : format === 'legacy'
    ? parseLegacy(parsed.value, DEFAULT_SECTIONS)
    : null;
  if (!preview) {
    setImportError('Unrecognized YAML format (missing sections or scores)');
    return;
  }
  setImportPreview(preview);
  importDialogRef.current?.showModal();
  e.target.value = ''; // allow re-import of same file
};

const handleImportConfirm = async (overwriteActive: boolean) => {
  if (!importPreview) return;
  await useAppStore.getState().importSession(importPreview.result, overwriteActive);
  setImportPreview(null);
};
```
